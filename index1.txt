    &lt;script&gt;
        const dropZone = document.getElementById('dropZone');
        const fileInput = document.getElementById('files');
        const fileList = document.getElementById('fileList');
        const btnAction = document.getElementById('btnAction');
        let selectedFiles = [];

        // Drag and Drop Logic
        ['dragenter', 'dragover', 'dragleave', 'drop'].forEach(eventName =&gt; {
            dropZone.addEventListener(eventName, e =&gt; { e.preventDefault(); e.stopPropagation(); }, false);
        });

        dropZone.addEventListener('drop', e =&gt; {
            handleFiles(e.dataTransfer.files);
        });

        dropZone.addEventListener('click', () =&gt; fileInput.click());
        fileInput.addEventListener('change', e =&gt; handleFiles(e.target.files));

        function handleFiles(files) {
            selectedFiles = [...files].filter(f =&gt; f.type === "application/pdf");
            fileList.innerHTML = '';
            selectedFiles.forEach(f =&gt; {
                const span = document.createElement('span');
                span.className = "bg-slate-700 px-3 py-1 rounded text-xs border border-slate-600";
                span.textContent = f.name;
                fileList.appendChild(span);
            });
            btnAction.disabled = selectedFiles.length === 0;
        }

        // Conversión Números a Letras Pro
        function numALetras(n) {
            const u = ['', 'UN', 'DOS', 'TRES', 'CUATRO', 'CINCO', 'SEIS', 'SIETE', 'OCHO', 'NUEVE'];
            const d = ['', 'DIEZ', 'VEINTE', 'TREINTA', 'CUARENTA', 'CINCUENTA', 'SESENTA', 'SETENTA', 'OCHENTA', 'NOVENTA'];
            const e = {11:'ONCE', 12:'DOCE', 13:'TRECE', 14:'CATORCE', 15:'QUINCE', 16:'DIECISÉIS', 17:'DIECISIETE', 18:'DIECIOCHO', 19:'DIECINUEVE'};
            
            if (n === 0) return 'CERO';
            if (n &lt; 10) return u[n];
            if (e[n]) return e[n];
            if (n &lt; 100) return d[Math.floor(n/10)] + (n%10 !== 0 ? ' Y ' + u[n%10] : '');
            return n.toString(); // Simplificado para el ejemplo
        }

        async function processFiles() {
            const { PDFDocument, StandardFonts, rgb } = PDFLib;
            const start = parseInt(document.getElementById('startNum').value);
            const format = document.getElementById('format').value;
            const fontName = document.getElementById('font').value;
            
            btnAction.textContent = "Procesando...";
            btnAction.disabled = true;

            const finalPdf = await PDFDocument.create();
            let currentFolio = start;

            for (const file of selectedFiles) {
                const bytes = await file.arrayBuffer();
                const sourcePdf = await PDFDocument.load(bytes);
                const pages = await finalPdf.copyPages(sourcePdf, sourcePdf.getPageIndices());
                const font = await finalPdf.embedFont(StandardFonts[fontName]);

                for (const page of pages) {
                    const { width, height } = page.getSize();
                    let text = '';
                    if (format === 'num') text = currentFolio.toString();
                    else if (format === 'text') text = numALetras(currentFolio);
                    else text = `${currentFolio} (${numALetras(currentFolio)})`;

                    page.drawText(text, {
                        x: width - 150,
                        y: height - 40,
                        size: 10,
                        font: font,
                        color: rgb(0.8, 0, 0)
                    });
                    finalPdf.addPage(page);
                    currentFolio++;
                }
            }

            const pdfBytes = await finalPdf.save();
            const blob = new Blob([pdfBytes], { type: 'application/pdf' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = "PROYECTO_FOLIADO.pdf";
            a.click();
            
            btnAction.textContent = "¡Listo!";
            setTimeout(() =&gt; { btnAction.textContent = "Procesar Archivos"; btnAction.disabled = false; }, 2000);
        }
